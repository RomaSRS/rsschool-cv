                      this.read = function () {
                        this.a = +prompt ('a?', '0');

                      },

                        return this.a + this.b;

                        return this.a * this.b
                    }
                  };
                  
                  let calculator = new Calculator();

                  alert( "Mul = " + calculator.mul() );
